func readLine(fileName string) []string {                            // é€è¡Œè¯»å–æ–‡ä»¶, è¿”å›å­—ç¬¦ä¸²åˆ‡ç‰‡, åˆ‡ç‰‡å…ƒç´ ç»“å°¾æ— æ¢è¡Œç¬¦å·

func readByte(fileName string, length int) string {                  // æ¯æ¬¡è¯»å–å›ºå®šé•¿åº¦å­—ç¬¦ä¸²

    content := []byte{}

        _, err := file.Read(buf)
        if err != nil { break }
        content = append(content, buf...)
    }
    return string(content)
